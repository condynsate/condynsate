    "start = time.perf_counter()\n",
    "print(\"Press 'esc' to end.\")\n",
    "while time.perf_counter()-start < 5.0:\n",
    "    # If the esc key is ever pressed, end the loop\n",
    "    if keyboard.is_pressed('esc'):\n",
    "        print(\"esc' pressed. Quitting.\")\n",
    "        break\n",
    "\n",
    "    # Get every pressed key. If any pressed keys are detected, print them\n",
    "    pressed = keyboard.get_pressed()\n",
    "    if len(pressed) > 0:\n",
    "        print(f\"Keys pressed: {pressed}\")\n",
    "\n",
    "    # Just to remove CPU strain\n",
    "    time.sleep(0.05)"
